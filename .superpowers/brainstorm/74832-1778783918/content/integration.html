6. DEBUG_FSM 로깅

const DEBUG_FSM := true

func transition_to(next: EnemyState, reason := "") -> void:
    if next == _current_state:
        if DEBUG_FSM:
            push_warning("transition_to(%s) but already in %s" % [next, _current_state])
        return
    if DEBUG_FSM:
        print("[FSM] %s -> %s (%s)" % [
            EnemyState.keys()[_current_state],
            EnemyState.keys()[next],
            reason
        ])
    _exit_state(_current_state)
    _current_state = next
    _state_enter_time = Time.get_ticks_msec()
    _enter_state(_current_state)
