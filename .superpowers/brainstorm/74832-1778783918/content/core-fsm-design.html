EnemyState enum

enum EnemyState {
    IDLE,    # 유일한 decision state
    ATTACK,  # 내부에 STARTUP/ACTIVE/RECOVERY sub-phase
    EVADE,   # 회피 이동
    HIT,     # 최우선 인터럽트 (피격/경직)
    DEAD     # KO 시퀀스, 종단 상태
}

transition_to() — 모든 전환의 단일 창구

var _current_state: EnemyState = EnemyState.IDLE

func transition_to(next: EnemyState) -> void:
    if next == _current_state:
        return
    _exit_state(_current_state)
    _current_state = next
    _enter_state(_current_state)

func _enter_state(state: EnemyState) -> void:
    match state:
        IDLE:
            _reset_timers()      # attack/evade idle timer만 초기화
            _select_next_action() # 바로 다음 행동 결정 가능
        ATTACK:
            _attack_phase = AttackPhase.STARTUP
            _start_attack_telegraph()
        EVADE:
            _start_evade_tween()
        HIT:
            _cancel_attack()     # 진행 중인 attack 정리
            _play_hit_animation()
        DEAD:
            _start_ko_sequence()

func _exit_state(state: EnemyState) -> void:
    match state:
        ATTACK:
            _cleanup_attack_timers()
        EVADE:
            _cleanup_evade_tween()
        HIT:
            # hit recovery 끝나면 자연스럽게
    # IDLE/DEAD: no cleanup needed

_process(delta) — 단일 디스패치

func _process(delta: float) -> void:
    if _current_state == EnemyState.DEAD:
        return
    if not ai_enabled:
        return
    match _current_state:
        EnemyState.IDLE:
            _update_idle(delta)
        EnemyState.ATTACK:
            _update_attack(delta)
        EnemyState.EVADE:
            _update_evade(delta)
        EnemyState.HIT:
            _update_hit(delta)

IDLE — 유일한 decision state

func _update_idle(delta: float) -> void:
    _attack_idle_timer -= delta
    _evade_idle_timer -= delta

    if _attack_idle_timer <= 0.0:
        transition_to(EnemyState.ATTACK)
        return
    if _evade_idle_timer <= 0.0:
        transition_to(EnemyState.EVADE)
        return
    # 둘 다 준비 안 됐으면 대기

# 타이머 리셋 시에만 난이도 스케일링 적용
func _reset_timers() -> void:
    _attack_idle_timer = randf_range(
        attack_delay_min * _difficulty_mult,
        attack_delay_max * _difficulty_mult
    )
    _evade_idle_timer = randf_range(
        evade_idle_min * _difficulty_mult,
        evade_idle_max * _difficulty_mult
    )
